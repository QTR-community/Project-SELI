fetch("/api/profiles/", {
  method: "POST",
  headers: {
    "Content-Type": "application/json",
    "X-CSRFToken": "{{ csrf_token }}"
  },
  body: JSON.stringify({
    user_id: 1,
    street: "123 Broad St",
    city: "Lagos",
    state: "Lagos"
  })
})
.then(res => res.json())
.then(data => console.log("Profile created:", data));
